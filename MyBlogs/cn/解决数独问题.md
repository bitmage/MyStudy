class Solution {
public:

    void display(vector<vector<char>> &board) {
        for (int idx = 0; idx < 9; idx++) {
            for (int idy = 0; idy < 9; idy++) {
                cout << board[idx][idy] << ' ';
            }
            cout << endl;
        }
        cout << endl;
    }

    bool solve(vector<set<char>> &rows, vector<set<char>> &cols, vector<vector<char>> &board) {
        // display(board);
        bool isOk = true;
        for (int idx = 0; idx < 9; idx++) {
            for (int idy = 0; idy < 9; idy++) {
                if (board[idx][idy] == '.') {
                    isOk = false;
                    for (char ch = '1'; ch <= '9'; ch++) {
                        if (rows[idx].find(ch) != rows[idx].end()
                                && cols[idy].find(ch) != cols[idy].end())
                        {
                            rows[idx].erase(ch);
                            cols[idy].erase(ch);
                            board[idx][idy] = ch;
                            isOk = isOk || solve(rows, cols, board);
                            if (isOk) {
                                return isOk;
                            }
                            board[idx][idy] = '.';
                            rows[idx].insert(ch);
                            cols[idy].insert(ch);
                        }
                    }
                }
            }
        }
        return isOk;
    }

    void solveSudoku(vector<vector<char>>& board) {
        vector<set<char>> rows(9), cols(9);
        for (int i = 0; i < 9; i++) {
            for (int c = 0; c < 9; c++) {
                rows[i].insert(c + '1');
                cols[i].insert(c + '1');
            }
        }

        for (int idx = 0; idx < 9; idx++) {
            for (int idy = 0; idy < 9; idy++) {
                if (board[idx][idy] != '.') {
                    rows[idx].erase(board[idx][idy]);
                    cols[idy].erase(board[idx][idy]);
                }
            }
        }

        solve(rows, cols, board);
    }
};
